/create the solution file
dotnet new sln

/create the controllers of the webapi
dotnet new webapi -controllers -n API

/add the api to the solution
dotnet sln add API

/change directory to API
cd API

/runs the webapp
dotnet run

dotnet dev-certs https --check
dotnet dev-certs https --clean
dotnet dev-certs https --trust

/restores any packages
dotnet restore


/create database
dotnet ef database update


/install the dotnet entity framework tool
dotnet tool install -g dotnet-ef

/create migration
dotnet ef migrations add InitialCreate -o Data/Migrations

dotnet new gitignore

/create an angular client folder
ng new client

/create a sapplication with styles but run this command from client
ng serve
